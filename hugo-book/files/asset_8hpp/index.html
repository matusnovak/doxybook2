    #pragma once
#include <queue>
#include <string>

namespace Engine {
    namespace Assets {
        class Asset {
        public:
            union Event {
                struct Loaded {
                    bool success;
                    int usage;
                } loaded;
                struct Unloaded {
                    bool success;
                } unloaded;
                struct Waiting {
                    Asset* other;
                } waiting;
            };

            explicit Asset(const std::string& name);
            virtual ~Asset() = 0;

            virtual void load();
            virtual void unload();
            const Event& getEvent() const {
                return event;
            }
        private:
            Event event;
        };
    }
}
